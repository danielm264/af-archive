function doPost(e) {
  try {
    // Set the main Google Drive folder ID
    var folderId = "1KgZonKFv7PfOk8X_UTg1aqz26OiE6KWC";
    var folder = DriveApp.getFolderById(folderId);

    // Extract all form fields from URL parameters
    var name = e.parameter.name || "Anonymous";
    var email = e.parameter.email || "No Email Provided";
    var phone = e.parameter.phone || "No Phone Provided";
    var applicationType = e.parameter.applicationType || "Not Specified";
    var designerDetails = e.parameter.designerDetails || "";
    var otherDetails = e.parameter.otherDetails || "";

    // Create timestamp (Brisbane time)
    var now = new Date();
    var formattedTimestamp = Utilities.formatDate(now, "Australia/Brisbane", "dd_MM_yyyy_HH_mm_ss");

    // Create subfolder with name and timestamp
    var subfolderName = name + "_" + formattedTimestamp;
    var subfolder = folder.createFolder(subfolderName);

    // Parse the POST body to get all file data
    var formData = JSON.parse(e.postData.contents);

    // Function to save a file from the data
    function saveFile(fileData, prefix) {
      if (fileData && fileData.name) {
        var blob = Utilities.newBlob(fileData.data, fileData.type, prefix + "_" + fileData.name);
        return subfolder.createFile(blob);
      }
      return null;
    }

    // Save all files
    var rationaleFile = saveFile(formData.files.rationale, "Rationale");
    var cvFile = saveFile(formData.files.cv, "CV");
    var headshotFile = saveFile(formData.files.headshot, "Headshot");
    var workFile = saveFile(formData.files.work, "Work");

    // Create and save submission details text file
    var submissionDetails = 
      "Submission Details\n" +
      "=================\n\n" +
      "Timestamp: " + formattedTimestamp + "\n\n" +
      "Personal Information:\n" +
      "-------------------\n" +
      "Name: " + name + "\n" +
      "Email: " + email + "\n" +
      "Phone: " + phone + "\n\n" +
      "Application Details:\n" +
      "------------------\n" +
      "Application Type: " + applicationType + "\n" +
      (designerDetails ? "Designer Details: " + designerDetails + "\n" : "") +
      (otherDetails ? "Other Details: " + otherDetails + "\n" : "") +
      "\nFiles Submitted:\n" +
      "--------------\n" +
      "Rationale: " + (rationaleFile ? rationaleFile.getName() : "Not provided") + "\n" +
      "CV: " + (cvFile ? cvFile.getName() : "Not provided") + "\n" +
      "Headshot: " + (headshotFile ? headshotFile.getName() : "Not provided") + "\n" +
      "Work Sample: " + (workFile ? workFile.getName() : "Not provided") + "\n";

    // Save details to text file
    var detailsFile = subfolder.createFile("submission_details.txt", submissionDetails);

    // Create email content with more detailed information
    var emailBody = 
      "A new Antifreeze submission has been received.\n\n" +
      "Applicant Information:\n" +
      "--------------------\n" +
      "Name: " + name + "\n" +
      "Email: " + email + "\n" +
      "Phone: " + phone + "\n" +
      "Application Type: " + applicationType + "\n" +
      (designerDetails ? "Designer Details: " + designerDetails + "\n" : "") +
      (otherDetails ? "Other Details: " + otherDetails + "\n" : "") +
      "\nFiles:\n" +
      "------\n" +
      "" + (rationaleFile ? rationaleFile.getName() : "Not Provided") + "\n" +
      "" + (cvFile ? cvFile.getName() : "Not provided") + "\n" +
      "" + (headshotFile ? headshotFile.getName() : "Not provided") + "\n" +
      "" + (workFile ? workFile.getName() : "Not provided") + "\n\n" +
      "Access submission folder at:\n" + subfolder.getUrl();

    // Send email notification
    MailApp.sendEmail({
      to: "dmendels01@gmail.com",
      subject: "New Antifreeze Submission - " + name,
      body: emailBody
    });

    // Return success response
    return ContentService
      .createTextOutput(JSON.stringify({ success: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    // Log error for debugging
    Logger.log(error);
    
    // Return error response
    return ContentService
      .createTextOutput(JSON.stringify({ error: error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
